"aabbcde" -> 2 # 'a' and 'b'
"aabBcde" -> 2 # 'a' occurs twice and 'b' twice (`b` and `B`)
"indivisibility" -> 1 # 'i' occurs six times
"aA11" -> 2 # 'a' and '1'
"ABBA" -> 2 # 'A' and 'B' each occur twice
						


					def duplicate_count(text):
						seen = set()
						dupes = set()
						for char in text:
							char = char.lower()
							if char in seen:
								dupes.add(char)
							seen.add(char)
						return len(dupes)
						


					def duplicate_count(text):
						count = 0
						for c in set(text.lower()):
							if text.lower().count(c) > 1:
								count += 1
						return count
